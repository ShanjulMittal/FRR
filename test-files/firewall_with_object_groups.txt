! Firewall Configuration with Object Groups for Module 2 Testing
! Contains GRP-PROD-SERVERS object group as required for TC-2.1
!
hostname test-firewall-module2
!
! Object group definitions
object-group network GRP-PROD-SERVERS
 description Production servers group
 network-object host 10.1.1.1
 network-object host 10.1.1.2
!
object-group network WEB_SERVERS
 network-object host 192.168.1.10
 network-object host 192.168.1.11
!
object-group service WEB_SERVICES tcp
 service-object tcp eq 80
 service-object tcp eq 443
!
! Access Control Lists using object groups
access-list OUTSIDE_IN line 1 extended permit tcp object-group GRP-PROD-SERVERS any eq 80
access-list OUTSIDE_IN line 2 extended permit tcp object-group GRP-PROD-SERVERS any eq 443
access-list OUTSIDE_IN line 3 extended deny tcp any object-group GRP-PROD-SERVERS eq 22
access-list OUTSIDE_IN line 4 extended permit tcp object-group WEB_SERVERS object-group GRP-PROD-SERVERS object-group WEB_SERVICES
!
! Rules with external IPs for TC-2.5 testing
access-list OUTSIDE_IN line 5 extended permit tcp host 8.8.8.8 any eq 53
access-list OUTSIDE_IN line 6 extended deny tcp any host 8.8.8.8 eq 80
!
! Mixed rules with both object groups and plain IPs
access-list INSIDE_OUT line 1 extended permit tcp host 10.1.1.1 any eq 3306
access-list INSIDE_OUT line 2 extended deny tcp object-group GRP-PROD-SERVERS host 8.8.4.4 eq 443
!
! End of configuration
